static List<List<string>> GroupStrings(List<string> input)
{
    List<List<string>> result = new List<List<string>>();
    Dictionary<string, bool> string_list_dict = CreateStrDict(input); // 為輸入的 List<string>, 建立一個 Dict, Value 全部為 false
    for (int i = 0; i < input.Count; i++)
    {
        List<string> same_char = new List<string>();
        string str = input[i];
        if (string_list_dict[str] == true) // 當 string_list_dict[str] == true 表示該 str 已經進行過判斷分類, 就跳過
        {
            continue;
        }
        string_list_dict[str] = true;
        same_char.Add(str);
        for (int j = i + 1; j < input.Count; j++) // 依序往後比較
        {
            bool compare = CompareTwoStr(str, input[j]); // 判斷兩個 str 是否一樣
            if (compare)
            {
                same_char.Add(input[j]); // 一樣就添加到 str 的 list 李
                string_list_dict[input[j]] = true; // 並將被比較過的 string_list_dict[input[j]] 改為 true, 下一個比較迴圈就會跳過他
            }
        }
        result.Add(same_char); // 將該 List 輸入到最終的 List
    }
    return result;
}

static bool CompareTwoStr(string str1, string str2)
{
    //bool result = true;
    int count = 0;
    Dictionary<char, int> str_dict = CountTextChar(str1); // 為第一個 string 建立 char 數量的 dict
    for (int i = 0; i < str2.Length; i++) // loop through 所有 string2 char
    {
        if (!str_dict.ContainsKey(str2[i])) // 如果沒有出現該 char key 表示不一樣 retrun false
        {
            return false;
        }
        if (str_dict.ContainsKey(str2[i])) // 每出現一次 char, 就在 dict 內搜尋該 key 並 -1
        {
            count = str_dict[str2[i]]--; 
        }
        if (count < 0) // 如果 char 數量太多減到 <0 一樣是不一樣, return false
        {
            return false;
        }
    }
    // 最後檢查 str1 所建立的 char dict, 裡面的 value 是不是都是 0
    return !str_dict.Values.Any(x => x != 0); // 此語法表示如果 str_dict 內的 value 如果有不為 0就 return true
}

static Dictionary<char, int> CountTextChar(string str)
{
    Dictionary<char, int> d = new Dictionary<char, int>();
    for (int i = 0; i < str.Length; i++) // loop through string 每個 char 存到 dict, 並計算每一個 char 出現的次數
    {
        if (d.ContainsKey(str[i]))
        {
            d[str[i]]++;
        }
        else
        {
            d[str[i]] = 1;
        }
    }
    return d;
}

static Dictionary<string, bool> CreateStrDict(List<string> strings)
{
    Dictionary<string, bool> d = new Dictionary<string, bool>(); // 建立一個空字典, key: vaule type 為 string:bool
    foreach (string s in strings) // 將 List<string> 內的所有內容分別取出, 輸入到 dict 
    {
        d[s] = false;
    }
    return d;
}
