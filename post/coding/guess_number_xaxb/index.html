使用 C# 語法

此篇文章要做一個交互式的猜數字遊戲, 就是常見的猜幾A幾B的遊戲

規則：

  • 從 0~9 取出不重複的數字組成數列, 隨機排列

  • 玩家輸入四個數字, 位置和數字同時正確為A, 位置錯誤數字正確為B

  • ex：題目 1638, 猜 1324, 給出提示為 1A1B

這裡會採用兩種寫法：

  • 「程序式寫法」(Procedural Approach)

    • 所有 logic 和 process 集中在一個主程式中, 從上到下順序執行, 沒有明顯的 module, 所有的 logic 都在同一個地方執行

  • 「模組化寫法」(Modular Approach)

    • 將程式碼拆分為多個獨立 function, 每個 function 負責特定功能, 之後再把他們組裝起來

Procedural Approach

bool again = true;

//int[] answers = { -1, -1, -1, -1 };
List<int> answers = new List<int>(); // 建立一個 list 來存放答案
int answer_length = 4;

for (int i = 0; i < answer_length; i++) // 透過 for 迴圈, 產生四個不重複的數字
{
    Random rnd = new Random(Guid.NewGuid().GetHashCode());
    int random_num = rnd.Next(0, 10); // 範圍 0~9
    while (answers.Contains(random_num)) // 當 list 裡面出現重複的數字, 重新產生
    {
        random_num = rnd.Next(0, 10);
    }
    answers.Add(random_num);
}

Console.Write(string.Join(",", answers)); // 這是為了測試用顯示答案
Console.WriteLine();

int[] guess_num = new int[4]; // 建立一個 array, 存放 user 輸入的數字
int A = 0;
int B = 0;
while (again) // 當 again 持續為 true 表示遊戲還未結束
{
    string input = "";
    bool re_input = true;
    List<int> check_duplicate = new List<int>();
    while (re_input)
    {
        re_input = false;
        input = Console.ReadLine();
        for (int i = 0; i < input.Length; i++) // 檢查 user 輸入的 string 是否都可以轉成數字
        {
            if (!int.TryParse(input[i].ToString(), out int num1))
            {
                Console.WriteLine("Error Syntax! Input again."); // 如果失敗, 顯示錯誤
                re_input = true; // re_input 變成 true
                break;
            }
        }
        if (re_input) // 當 re_input 為 true, 跳過後續所有的 code, 要求 user 重新輸入
        {
            continue;
        }
        if (input.Length != 4) // 檢查 user 輸入的字串是否為4
        {
            re_input = true;
            Console.WriteLine("Input 4 numbers!");
            continue;
        }
        if (re_input)
        {
            continue;
        }
        for (int i = 0; i < input.Length; i++) // 檢查 user 輸入的 string 是否重複
        {
            if (check_duplicate.Contains(input[i]))
            {
                Console.WriteLine("Duplicate num! Input again.");
                re_input = true;
                check_duplicate = new List<int>();
                break;
            }
            else
            {
                check_duplicate.Add(input[i]);
            }
        }
    }
    for (int i = 0; i < input.Length; i++) // 將 user 的輸入, 存到 int array input_num中方便後續比對
    {
        int input_num = int.Parse(input[i].ToString());
        guess_num[i] = input_num;
    }
    for (int i = 0; i < guess_num.Length; i++) // loop through 輸入的數字
    {
        if (guess_num[i] == answers[i]) // 當 input_num[i] 剛好也等於 answers[i], 表示數字和位置都對 A++
        {
            A++;
            continue;
        }
        if (guess_num[i] != answers[i] && answers.Contains(guess_num[i])) // 當 input_num[i] 位置不等於 answers[i], 但有包含在 answers 內, 表示數字對位置不對 B++
        {
            B++;
            continue;
        }
    }
    Console.WriteLine($"{A} A {B} B");
    if (A == 4) // 當 A = 4, 表示 input 和 anserts 皆完整一樣, 遊戲結束
    {
        again = false; // again 等於 false 跳出迴圈
    }
    A = 0;
    B = 0;
}

static void Main(string[] args)
{
    int num = 4;
    List<int> answers = GenerateNonRepeateQuantityNumberZeroToNine(num);
    Console.Write(string.Join(",", answers));
    Console.WriteLine();

    while (true)
    {
        string input = Console.ReadLine();
        List<string> input_list = ConvertStringToStringList(input); // string 轉成 string list
        if (CheckNumSyntaxCorrect(input_list) == false) // 檢查語法
        {
            continue;
        }
        if (CheckInputNumQuantity(input_list, num) == false) // 檢查輸入數量
        {
            continue;
        }
        if (CheckDuplicate(input_list) == true) // 檢查是否重複
        {
            continue;
        }
        var results = GetGuessNumber(answers, input_list);
        Console.WriteLine($"{results.Item1} A {results.Item2} B");
        if (results.Item1 == num) // 如果 return A 的數量和 num 一樣都是 4 遊戲結束
        {
            break;
        }
    }
}

static List<string> ConvertStringToStringList(string input) // 將 string 轉換成 string list, return List<string>
{
    List<string> result = new List<string>();
    for (int i = 0; i < input.Length; i++)
    {
        result.Add(input[i].ToString());
    }
    return result;
}

static List<int> GenerateNonRepeateQuantityNumberZeroToNine(int quantity) // 產生一個n個不重複數字的 int List, return List<int>
{
    List<int> number_list = new List<int>();
    //int answer_length = 4;
    for (int i = 0; i < quantity; i++)
    {
        Random rnd = new Random(Guid.NewGuid().GetHashCode());
        int random_num = rnd.Next(0, 10);
        while (number_list.Contains(random_num))
        {
            random_num = rnd.Next(0, 10);
        }
        number_list.Add(random_num);
    }
    return number_list;
}

static bool CheckDuplicate(List<string> input) // 檢查 string List 的內容是否皆不重複, return bool
{
    List<int> check_duplicate = new List<int>();
    for (int i = 0; i < input.Count; i++)
    {
        if (check_duplicate.Contains(int.Parse(input[i])))
        {
            check_duplicate = new List<int>();
            Console.WriteLine("Duplicate num!");
            return true;
        }
        else
        {
            check_duplicate.Add(int.Parse(input[i]));
        }
    }
    return false;
}

static bool CheckNumSyntaxCorrect(List<string> input) // 檢查 string List 的內容是否能轉換成 int, return bool
{
    for (int i = 0; i < input.Count; i++)
    {
        if (!int.TryParse(input[i].ToString(), out int num1))
        {
            Console.WriteLine("Error Syntax! Input again.");
            return false;
        }
    }
    return true;
}

static bool CheckInputNumQuantity(List<string> input, int num) // 檢查 string list, 是否和所需要的 number 個是相符, return bool
{
    if (input.Count != num)
    {
        Console.WriteLine($"Input {num} numbers!");
        return false;
    }
    return true;
}

static (int, int) GetGuessNumber(List<int> answers, List<string> input) // 輸入兩個 list(int, string), 進行比較完成猜數字遊戲, return (int, int)
{
    int A = 0;
    int B = 0;
    for (int i = 0; i < input.Count; i++)
    {
        if (int.Parse(input[i].ToString()) == answers[i])
        {
            A++;
            continue;
        }
        if (int.Parse(input[i].ToString()) != answers[i] && answers.Contains(int.Parse(input[i].ToString())))
        {
            B++;
            continue;
        }
    }
    return (A, B);
}
