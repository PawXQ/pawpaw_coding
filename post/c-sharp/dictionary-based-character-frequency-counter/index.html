static List<char> FreqencyChar(string str)
{
    List<char> c = new List<char>(); // 建立一個 dynamic array
    int count = 0;
    Dictionary<char, int> dic = new Dictionary<char, int>(); // 建立一個 dict, key value type 為 char, int
    foreach (char character in str) // loop str 所有的 char
    {
        if (dic.ContainsKey(character)) // 如果已有該 char 的 Key, value(計數) + 1
        {
            dic[character]++;
        }
        if (!dic.ContainsKey(character)) // 如果沒有該 char 的 Key, value = 1
        {
            dic[character] = 1;
        }
    }
    foreach (var item in dic) // loop dict 所有的 item
    {
        if (item.Value > count) // 如果 char 的 value(計數) 比較多, 替換掉最大計數(count)
        {
            count = item.Value;
            c.Clear(); // 清空 lisy
            c.Add(item.Key); // 添加該 key(char)
        }
        else if (item.Value == count) //如果 char 的 value(計數) 一樣多
        {
            c.Add(item.Key); // 添加該 key(char)
        }
    }
    return c;
}
