1. Two Sum

Topic

給一組數字 array, 和 target 數字, 返回總合為 target 的數字位置(indices), 不一定要第一組

不能使用同一個 indicies 兩次

解法：

  1. target - number[i] = ans, 如果 ans 有包含在 dict 的 key 裡面, 表示這兩個 number[i] 和 ans 總合為 target

  2. target - number[i] = ans, 如果 ans 不在 dict 內, 就添加當前的 number[i] 添加到 dict 裡面, key = number[i], value = i

public class Solution {
    public int[] TwoSum(int[] nums, int target) {
        int[] result = TwoSumCal(nums, target);
        return result;
    }
    // static
}

static int[] TwoSumCal(int[] numbers, int target)
{
    int[] result = new int[2]; // 創建總數為 2 的 array
    Dictionary<int, int> d = new Dictionary<int, int>(); // 建立 key, value 都為 int 的 dict
    // int j = 0;
    for (int i = 0; i < numbers.Length; i++)
    {
        int ans = target - numbers[i]; 
        if (d.ContainsKey(ans))
        {
            result[0] = d[ans];
            result[1] = i;
            return result;
        }
        else
        {
            d[numbers[i]] = i;
            // d[numbers[i]] = j;
            // j++;
        }
    }
    return result;
}
