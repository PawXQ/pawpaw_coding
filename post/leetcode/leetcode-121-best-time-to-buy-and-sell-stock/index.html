121. Best Time to Buy and Sell Stock

Topic

股票買賣時機

給一組價格 array, 判斷在哪一天賣出會得到最高獲利, 不能在同一天買賣.

如果沒有最大獲利, 則最高獲利為 0

重點：

  1. 當天股票只要出現最低價, 當天就不會買賣

  2. 股票不能往前賣, 只能往後賣

static int MaxProfitCal(int[] price)
{
    int max_profit = 0;
    int min_price = price[0];
    for (int i = 0; i < price.Length; i++)
    {
        if (price[i] < min_price)
        {
            min_price = price[i];
        }
        else if (price[i] - min_price > max_profit)
        {
            max_profit = price[i] - min_price;
        }
    }
    return max_profit;
}

static int[] MaxProfitDate(int[] price)
{
    int[] date = new int[2];
    int max_profit = 0;
    int min_price = price[0];
    int min_date = 0;
    for (int i = 0; i < price.Length; i++)
    {
        if (price[i] < min_price)
        {
            min_price = price[i];
            min_date = i + 1;
        }
        else if (price[i] - min_price > max_profit)
        {
            max_profit = price[i] - min_price;
            date[0] = min_date;
            date[1] = i + 1;
        }
    }
    return date;
}

static int[] MaxLossDate(int[] price)
{
    int[] date = new int[2];
    int max_loss = 0;
    int max_price = 0;
    int max_date = 0;
    for (int i = 0; i < price.Length; i++)
    {
        if (price[i] > max_price)
        {
            max_price = price[i];
            max_date = i + 1;
        }
        else if (max_price - price[i] > max_loss)
        {
            max_loss = max_price - price[i];
            date[0] = max_date;
            date[1] = i + 1;
        }
    }
    return date;
}
