static int RenewMinCoin(int[] coin_value, int coin)
{
    int[] initial_value = InitialValue(coin); // 初始化一個 array initial_value, 長度為 coin + 1，所有元素初始值為 -1
    for (int i = 1; i <= coin; i++) // 從 1 到 coin loop 每一個金額
    {
        for (int j = 0; j < coin_value.Length; j++) // loop 每一個硬幣面值
        {
            // 檢查目前硬幣面值 coin_value[j] 是否小於等於目前目標金額 i
            // 且 initial_value[i - coin_value[j]] 不等於 -1，表示可以用該硬幣面值構成目前金額
            if (coin_value[j] <= i && initial_value[i - coin_value[j]] != -1)
                if (coin_value[j] <= i)
                {
                    // 計算減去目前硬幣面額後的剩餘金額
                    int minus = i - coin_value[j];
                    // 取得剩餘金額的最少硬幣數
                    int minus_value = initial_value[minus];
                    // 如果目前金額的最少硬幣數尚未計算過（值為 -1）
                    // 或目前金額的最少硬幣數大於減去目前硬幣面額後的最少硬幣數加 1
                    if (initial_value[i] == -1 || initial_value[i] > minus_value + 1)
                    {
                        initial_value[i] = minus_value + 1; // 更新目前金額的最少硬幣數為減去目前硬幣面額後的最少硬幣數加 1
                    }
                }
        }
    }
    return initial_value[coin];
}

static int[] InitialValue(int number)
{
    // 建立一個長度為 number + 1 的陣列
    int[] result = new int[number + 1];
    
    // 使用 for 迴圈初始化數組中的元素
    for (int i = 1; i <= number; i++)
    {
        result[i] = -1; // 將每個元素設為 -1
    }
    
    // 返回初始化后的数组
    return result;
}

貪婪演算法

這邊補充使用貪婪演算法計算的方式, 雖然這種方式在多數情況下答案並不正確, 但是是最簡易計算的方式

貪婪演算法就是尋找當下的最優解

適合場景, 每一個問題都可以拆分成子問題, 每一個子問題的最優解組合起來是問題的最優解

不過當下最優解, 不一定是問題最佳解

找錢來說, 理論上用幣值最大的下去做找零, 可以得到最少量的紙幣

但是在例如題目14 幣值10, 7, 2, 1 來說

當下最佳解會用最大的幣值1個10元

剩下4元用2個2元

總共3個硬幣

但其實2個7元, 才會是最少的硬幣

namespace <namespace_name>
{
    internal class Program
    {
        static void Main(string[] args)
        {
            //硬幣兌換: 任意輸入一組數字，找出該數字最少可用的硬幣
            // 9321
            // 幣值 1000,500,100,50,10,5,1
            // 1000有9個
            // 100有3個
            // 10有2
            // 1有1

            // 14
            // 幣值 1,2,7,10
            // 10元1個
            // 2元2個

            //int[] coin = { 1000, 500, 100, 50, 10, 5, 1 };
            int[] coin =  { 10, 7, 2 };
            int num = 15;
            int[] result = CoinExchange(coin, num);
            Console.WriteLine($"{num}");
            for (int i = 0; i < result.Length; i++)
            {
                if (result[i] == -1)
                {
                    Console.WriteLine("Unable to convert");
                }
                int coin_value = coin[i];
                int quantity = result[i];
                if (quantity > 0)
                {
                    Console.WriteLine($"{coin_value} dollar are {quantity}");
                }
            }
        }

        static int[] CoinExchange(int[] coin_value, int number)
        {

            int[] coin_quantity = new int[coin_value.Length];

            int num = 0;

            for (int i = 0; i < coin_value.Length; i++)
            {
                num = number / coin_value[i];
                coin_quantity[i] = num;
                number -= num * coin_value[i];
            }

            if (number != 0)
            {
                return new int[] { -1 };
            }
            return coin_quantity;
        }

    }
}
