e.g. [“1”,“2”,“Fizz”,“4”,“Buzz”,“Fizz”,“7”,“8”,“Fizz”,“Buzz”,“11”,“Fizz”,“13”,“14”,“FizzBuzz”]

15 因為可以同時被 3 和 5 整除所以會變成 FizzBuzz

static string[] FizzBuzz(int number)
{
    string[] result = new string[number]; // 宣告一個 string array, string array 在沒有給定 value 之前, 所有 value 都是 null
    for (int i = 0; i < number; i++)
    {
        if ((i + 1) % 3 == 0)
        {
            result[i] += "Fizz";
        }
        if ((i + 1) % 5 == 0) // 這邊使用 +=, 讓例如是 15 可以先判斷 3 是否整除變成 Fizz, 之後判斷 5 是否整除就可以變成 FizzBuzz, 而不是直接取代變成只有 Buzz
        {
            result[i] += "Buzz";
        }
        if (result[i] == null)
            result[i] = (i + 1).ToString();
    }
    return result;
}
