        return un_repeat_num;

static List<char> Create_char_List(string input)

    for (int i = 0; i < input.Length; i++)
    {
        char_list.Add(input[i]);
    }
    return char_list;

static bool CheckDuplicate(string str, char character)
{
    bool result = false;
    foreach (char c in str)
    {
        if (c == character) return true;
    }
    return result;
}

//static int FindMaxNonRepeatStringCount(string input) // 單純返回 長度
static (int, List<string>) FindMaxNonRepeatStringCount(string input)
{
    List<string> result = new List<string>();
    List<char> char_list = Create_char_List(input);
    int char_num = 0;
    for (int i = 0; i < input.Length; i++)
    {
        // 提前終止條件：如果剩餘 char 數不可能超過目前最大長度, 跳出循環
        if (input.Length - i <= char_num)
        {
            break;
        }
                
        int count = 0;
        string char_connect = "";
        for (int j = i; j < char_list.Count; j++)
        {
            if (CheckDuplicate(char_connect, char_list[j]))
            {
                break;
            }
            char_connect += char_list[j];
            count++;
            if (count > char_num)
            {
                char_num = count;
                result.Clear();
                result.Add(char_connect);
            }
            else if (count == char_num && !result.Contains(char_connect)) // 如果長度一樣, 確認是否重複再添加到 List
            {
                result.Add(char_connect);
            }
        }
    }
    //return char_num; // 單純返回 長度
    return (char_num, result);
}
