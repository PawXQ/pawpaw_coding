202. Happy Number

這篇會使用 C# 和 python 來檢查數字 Happy Number(快樂數)

什麼是快樂數!?

wiki - Happy Number

簡單說就是一個數字的所有位數的<數字平方和>相加等於 1, 如果相加沒有等於1 , 就再次進行所有位數的<數字平方和>, 重複進行直至最後等於 1

e.g.

28 -> 22 + 82 = 68 -> 62 + 82 = 100 -> 12 + 02 + 02 = 1

37 -> 32 + 72 = 58 -> 52 + 82 = 89 -> 82 + 92 = 145 -> 12 + 42 + 52 = 42 -> 42 + 22 = 20 -> 22 + 02 = 4 -> 42 = 16 -> 12 + 62 = 37…..

因此 28 是 Happy Number, 37 則不是 Happy Number

所有不是快樂數的所有的位數<數字平方和>都會進入 4 → 16 → 37 → 58 → 89 → 145 → 42 → 20 → 4 的循環

static bool Check_Happy(int number)
{
    Dictionary<int, int> d = new Dictionary<int, int>(); // 設定 dict d, key 和 value type 皆為 int
    bool check = true; 
    while (check) // 當 chcek 為 Ture (持續檢查)
    {
        int total = 0;
        while (number != 0)
        {
            int num = number % 10;
            total += num * num;
            number /= 10;
        }
        // int digit_100 = number / 100 % 10; // 取百位數
        // int digit_10 = number / 10 % 10; // 取十位數
        // int digit_1 = number / 1 % 10; // 取個位數
        // int total = digit_100 * digit_100 + digit_10 * digit_10 + digit_1 * digit_1; // 所有位數平方和
        if (total == 1) // 如果 = 1 就是 Happy Number
        {
            break; // 跳出迴圈
        }
        // 如果 dict d 裡面沒有該 <平方和> 的 key 則存入 dict d
        if (!d.ContainsKey(total)) // 判斷 dict d 裡面沒有該 <平方和> 的 key
        {
            d.Add(total, i); // 沒有則存入 { total , i} 
            i++;
            number = total; // 下一個數字為剛剛<所有位數的平方和>
            continue; // 下一個迴圈
        }
        if (d.ContainsKey(total)) // 判斷 dict d 裡面有<所有位數的平方和>
        {
            check = false; // 表示重複數字, 陷入循環一定不是 Happy Number, check = false 就不需再檢查
        }
    }
    return check;
}

static bool Check_Happy(int number) // Function 設定返還 bool, input 限定為 int
{
    HashSet<int> hash_int = new HashSet<int>(); // 設定 HashSet hash_int, value type 為 int
    bool check = true; 
    while (check) // 當 chcek 為 Ture (持續檢查)
    {
        int total = 0;
        while (number != 0) // 當輸入的 number 不為0
        {
            int num = number % 10; // 當前數字取餘數(個位數)
            total += num * num; // 最後一個數的平方
            number /= 10; // 除法會自動去除小數點, 因此假如是 123 / 10 = 12, 因此 12 % 10 = 2 就是十位數, 以此類推
        }
        // int digit_100 = number / 100 % 10; // 取百位數
        // int digit_10 = number / 10 % 10; // 取十位數
        // int digit_1 = number / 1 % 10; // 取個位數
        // int total = digit_100 * digit_100 + digit_10 * digit_10 + digit_1 * digit_1; // 所有位數平方和
        if (total == 1) // 如果 = 1 就是 Happy Number
        {
            break; // 跳出迴圈
        }
        // 如果 HashSet hash_int 裡面沒有該 <平方和> 的 value 則存入
        if (!hash_int.Contains(total)) // 判斷 HashSet hash_int 裡面沒有該 <平方和> 的 value
        {
            hash_int.Add(total); // 沒有則存入 { total } 
            number = total; // 下一個繼續運算的數字為剛剛<所有位數的平方和>
            continue; // 下一個迴圈
        }
        if (hash_int.Contains(total)) // 判斷 HashSet hash_int 裡面有<所有位數的平方和>
        {
            check = false; // 表示重複數字, 陷入循環一定不是 Happy Number, check = false 就不需再檢查
        }
    }
    return check;
}
