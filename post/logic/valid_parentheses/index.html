Description

此篇 coding 是在 C# 語言下編寫, 不過是考邏輯, 確認 括號 是否是有效使用

括號 需要是以下條件才算是有效

  1. 配對性：每個開括號必須有一個對應的閉括號

    • 簡單說就是有 左 就要有 右

    • e.g. 有 [ 就要有 ]

  2. 嵌套順序：括號必須按正確的順序嵌套，即內部的括號必須在外部括號對閉合之前閉合

    • correct：[()]

    • wrong [(]) 錯誤的可以看到內部 () 還會閉合前, 外部 [] 已經先閉合

  3. 閉合順序：在任何時刻，閉括號不能出現在無對應的開括號之前

    • 簡單說就是沒有 左 之前不能有 右

    • e.g. 沒有 [ 不能有 ]

static bool CheckBracketIsVaild(string input)
{
    Stack<char> stack = new Stack<char>();
    Dictionary<char, char> dict = new Dictionary<char, char>();

    dict.Add(')', '(');
    dict.Add(']', '[');
    dict.Add('}', '{');

    char[] symbol = { '(', '[', '{' }; // 用於存放所有左括號的 char array
    for (int i = 0; i < input.Length; i++)
    {
        if (symbol.Contains(input[i])) // 如果當前 char 在 symbol array 中, 則將其 push 到 stack 中
        {
            stack.Push(input[i]);
        }
        else // 如果當前 char 不是左括號, 則認為它是右括號. 此時需要從 stack 中 pop 一個左括號並進行匹配檢查
        {
            if (stack.Count == 0) //  如果 stack 為空, 說明沒有匹配的左括號, 返回 false
            {
                return false;
            }
            char c = stack.Pop();
            if (dict[input[i]] != c) // 先 pop stack 的左括號, 使用字典 dict 來檢查當前的右括號是否與 pop 的左括號匹配. 如果不匹配, 則返回 false，表示括號無效
            {
                return false;
            }
        }
    }
    return stack.Count == 0; // 檢查堆疊是否已清空
}
