string asterisk = "*";
int row = 5;

for (int i = 1; i <= row; i++) // 第n個 row
{
    for (int j = 1; j <= 2 * i - 1; j++) // 例如第一個 row 就會是 2 * 1 - 1 只需要 output 一個 *
    {
        Console.Write(asterisk);
    }
    Console.WriteLine(); // 換行輸出
}

// 上半部
for (int i = 1; i <= row; i++) // 第 n 個 row
{
    for (int j = 1; j <= row - i; j++) // 輸出 space, space 數量為 total_row - row
    {
        Console.Write(" ");
    }
    for (int k = 1; k <= 2 * i - 1; k++) // 輸出 "*"
    {
        Console.Write(asterisk);
    }
    Console.WriteLine();
}
// 下半部
for (int i = 1; i <= row; i++)
{
    for (int j = 1; j < i; j++) // 輸出 space, space 數量為 row -1
    {
        Console.Write(" ");
    }
    for (int k = 1; k <= 2 * (row - i) + 1; k++) // 輸出 "*"
    {
        Console.Write(asterisk);
    }
    Console.WriteLine();
}
