import tkinter
import random
import pandas


BACKGROUND_COLOR = "#B1DDC6"
WORKING_DICT = "flash-card-project-start"
current_vocab = {}
flash_timer = ""

# ---------------------------- FUNCTION ------------------------------- #


def random_choose_vocabulary():
    global current_vocab
    random_vocabulary = random.choice(data_vocabulary_list)
    current_vocab = random_vocabulary


def flash_french_vocabulary():
    global flash_timer
    if flash_timer:
        window.after_cancel(flash_timer)
    random_choose_vocabulary()
    canvas.itemconfig(canvas_bg_image, image=card_front_image)
    canvas.itemconfig(cnt_text, text="French", fill="black")
    canvas.itemconfig(vocab_text, text=current_vocab["French"], fill="black")
    flash_timer = window.after(3000, flash_eng_vocabulary)


def flash_eng_vocabulary():
    canvas.itemconfig(canvas_bg_image, image=card_back_image)
    canvas.itemconfig(cnt_text, text="English", fill="white")
    canvas.itemconfig(vocab_text, text=current_vocab["English"], fill="white")


def known_card():
    data_vocabulary_list.remove(current_vocab)
    data = pandas.DataFrame(data_vocabulary_list)
    data.to_csv(f"{WORKING_DICT}/data/french_words_to_learn.csv", index=False)
    flash_french_vocabulary()

# ---------------------------- DATA ----------------------------------- #


try:
    data_vocabulary = pandas.read_csv(f"{WORKING_DICT}/data/french_words_to_learn.csv")
except FileNotFoundError:
    data_vocabulary = pandas.read_csv(f"{WORKING_DICT}/data/french_words.csv")
    data_vocabulary_list = data_vocabulary.to_dict(orient="records")
else:
    data_vocabulary_list = data_vocabulary.to_dict(orient="records")


# ---------------------------- UI SETUP ------------------------------- #


window = tkinter.Tk()
window.title("Flashy")
window.config(padx=50, pady=50, bg=BACKGROUND_COLOR)


card_back_image = tkinter.PhotoImage(file=f"{WORKING_DICT}/images/card_back.png")
card_front_image = tkinter.PhotoImage(file=f"{WORKING_DICT}/images/card_front.png")
right_image = tkinter.PhotoImage(file=f"{WORKING_DICT}/images/right.png")
wrong_image = tkinter.PhotoImage(file=f"{WORKING_DICT}/images/wrong.png")


canvas = tkinter.Canvas(width=800, height=526, bg=BACKGROUND_COLOR, highlightthickness=0)
canvas_bg_image = canvas.create_image(400, 263, image=card_front_image)
canvas.grid(row=0, column=0, columnspan=2)

cnt_text = canvas.create_text(400, 150, text="", font=("Ariel", 40, "italic"))
vocab_text = canvas.create_text(400, 253, text="", font=("Ariel", 60, "bold"))

button_right = tkinter.Button(image=right_image, highlightthickness=0, command=known_card)
button_right.grid(row=1, column=0)
button_wrong = tkinter.Button(image=wrong_image, highlightthickness=0, command=flash_french_vocabulary)
button_wrong.grid(row=1, column=1)

flash_french_vocabulary()

window.mainloop()
