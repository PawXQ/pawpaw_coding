Return

  • Function 可以有多個 return statement

  • return 表示 function 結束

  • return 之後的 code 都不會執行

  • 使用空的 return 相當於 return None

import os

def add(num1, num2):
    return num1 + num2

def subtract(num1, num2):
    return num1 - num2

def multiply(num1, num2):
    return num1 * num2

def divide(num1, num2):
    return num1 / num2

operations = {
    '+' : add,
    '-' : subtract,
    '*' : multiply,
    '/' : divide,
}

def get_operation():
    operation_symbol = input("Pick an operation: ")
    while operation_symbol not in operations:
        operation_symbol = input("Invalid input. Please type '+', '-', '*', '/': ")
    return operation_symbol

def get_num(promt):
     return float(input(promt))

def continue_calculate(result):
    again = input(f"Type 'y' to continue calculating with {result}, or 'n' to start a new calculation: ").lower()
    while again not in ["yes", "y", "no", "n"]:
        again = input("Invalid input. Please type 'y', 'n': ").lower()
    return again in ['y', 'yes']


def calculate():
    while(True):
        num1 = get_num("What's the first number?: ")

        while(True):

            for i in operations:
                print(i)

            operation_symbol = get_operation()

            num2 = get_num("What's the next number?: ")

            result = operations[operation_symbol](num1, num2)

            print(f"{num1} {operation_symbol} {num2} = {result}")

            num1 = result

            if not continue_calculate(result):
                break

calculate()
