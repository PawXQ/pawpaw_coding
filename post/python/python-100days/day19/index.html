Multiple Instance

  • 使用 class 可以建立出多個 instance

  • 每一個 instance 都是獨立的 object

  • instance 在記憶體中有各自存放的位置

  • 每個實例擁有獨立的 state, 包括 attributes 和 methods

特點

  • instance 間相互獨立

  • 可以有不同的 attribute

  • method 調用互不影響

# Create two independent Turtle instances

tom = turtle.Turtle()

print(tim.pencolor())  # output: green
print(tom.pencolor())  # output: blue

import turtle

all_turtle = []

for i in range(2):
    new_turtle = turtle.Turtle()
    new_turtle.color("black")
    all_turtle.append(new_turtle)

# Modify the color of the first instance
all_turtle[0].color("yellow")

# Verify the independence of each instance
print(all_turtle[0].pencolor())  # result: yellow
print(all_turtle[1].pencolor())  # result: black

import turtle
import random


def turtle_race():
    for turtle in turtle_list:
        if turtle.xcor() > 200:
            winning_color = turtle.pencolor()
            if winning_color == user_bet:
                print(f"You win!")
            else:
                print(f"You lose winning color is {winning_color}.")
            return False

        random_move = random.randint(0, 10)
        turtle.fd(random_move)
    return True


colors = ["red", "orange", "yellow", "green", "blue", "purple"]
distance = 300/(len(colors)-1)
turtle_list = []


screen = turtle.Screen()
screen.setup(width=500, height=400)
screen.bgcolor("black")
user_bet = screen.textinput(title="Make your bet", prompt="Which turtle willl win the race? Enter a color: ")

for i in range(len(colors)):
    new_turtle = turtle.Turtle("turtle")
    new_turtle.penup()
    new_turtle.color(colors[i])
    new_turtle.goto(-230, -150 + i * distance)
    turtle_list.append(new_turtle)

race_is_on = False

if user_bet:
    race_is_on = True

while race_is_on:
    race_is_on = turtle_race()

screen.exitonclick()
