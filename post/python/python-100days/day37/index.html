request module

HTTP 請求主要分為四種類型：

  • GET: 從外部系統獲取數據
  • POST: 發送數據到外部系統
  • PUT: 更新外部系統中的數據
  • DELETE: 刪除外部系統中的數據

# POST request

在使用 requests 模組進行 API 驗證時，通常將憑證資訊(如 token 或 secret)存放於 headers 中，這種方式相對於將敏感資訊放在 params 或 data 中更為安全，因為：

  1. 敏感資訊不會出現在 URL 中，避免了暴露風險(例如通過瀏覽器的歷史記錄或伺服器日誌)
  2. 將驗證資訊與數據分離，更符合 RESTful API 的設計原則

import requests
from datetime import datetime

# Configuration
USERNAME = "aaabbbccc"
TOKEN = "qqqaaazzz"
BASE_URL = "https://pixe.la/v1"

HEADERS = {
    "X-USER-TOKEN": TOKEN
}


def get_formatted_date(date=None):
    if date is None:
        date = datetime.now()
    return date.strftime("%Y%m%d")


def create_user():
    endpoint = f"{BASE_URL}/users"
    params = {
        "token": TOKEN,
        "username": USERNAME,
        "agreeTermsOfService": "yes",
        "notMinor": "yes"
    }

    response = requests.post(url=endpoint, json=params)
    print(f"Create User - Status: {response.status_code}")
    print(f"Response: {response.text}")
    return response


def create_graph(graph_id, name, unit):
    endpoint = f"{BASE_URL}/users/{USERNAME}/graphs"
    params = {
        "id": graph_id,
        "name": name,
        "unit": unit,
        "type": "float",
        "color": "ajisai"
    }

    response = requests.post(url=endpoint, json=params, headers=HEADERS)
    print(f"Create Graph - Status: {response.status_code}")
    print(f"URL: {response.url}")
    print(f"Response: {response.text}")
    return response


def post_pixel(graph_id, quantity):
    endpoint = f"{BASE_URL}/users/{USERNAME}/graphs/{graph_id}"
    params = {
        "date": get_formatted_date(),
        "quantity": str(quantity)
    }

    response = requests.post(url=endpoint, json=params, headers=HEADERS)
    print(f"Post Pixel - Status: {response.status_code}")
    print(f"URL: {response.url}")
    print(f"Response: {response.text}")
    return response


def update_pixel(graph_id, quantity):
    date = get_formatted_date()
    endpoint = f"{BASE_URL}/users/{USERNAME}/graphs/{graph_id}/{date}"
    params = {
        "quantity": str(quantity)
    }

    response = requests.put(url=endpoint, json=params, headers=HEADERS)
    print(f"Update Pixel - Status: {response.status_code}")
    print(f"URL: {response.url}")
    print(f"Response: {response.text}")
    return response


def delete_pixel(graph_id):
    date = get_formatted_date()
    endpoint = f"{BASE_URL}/users/{USERNAME}/graphs/{graph_id}/{date}"

    response = requests.delete(url=endpoint, headers=HEADERS)
    print(f"Delete Pixel - Status: {response.status_code}")
    print(f"URL: {response.url}")
    print(f"Response: {response.text}")
    return response


def delete_user():
    endpoint = f"{BASE_URL}/users/{USERNAME}"

    response = requests.delete(url=endpoint, headers=HEADERS)
    print(f"Delete Pixel - Status: {response.status_code}")
    print(f"URL: {response.url}")
    print(f"Response: {response.text}")
    return response


# Create user
create_user()

# Create new graph
create_graph(
    graph_id="graph1",
    name="Cycling Graph",
    unit="Km"
)

# Post pixel data
post_pixel(
    graph_id="graph1",
    quantity=1.5
)

# Update pixel
update_pixel(
    graph_id="graph1",
    quantity=3.0
)

# Delete pixel
delete_pixel(graph_id="graph1")

# Delete user
delete_user()
