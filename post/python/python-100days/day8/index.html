Positional Arguments

  • 調用函數時, parameter 的順序與定義時的順序相對應

  • ex：greet_with("Jack Bauer", "Nowhere")

Keyword Arguments

  • 通過指定參數名來傳遞參數

  • syntax：function_name(parameter=value, parameter=value)

  • ex：greet_with(name="Angela", location="London")

  • 使用關鍵字參數可以避免位置參數的順序問題

import art

alphabet = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l',
            'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z']


def encrypt(original_text, shift_amount):
    new_text = ""
    len_alphabet = len(alphabet)
    for char in original_text:
        if char in alphabet:
            char_index = alphabet.index(char)
            word_shift = alphabet[(char_index+shift_amount) % len_alphabet]
            new_text += word_shift
        else:
            new_text += char
    print(f"Here is the encoded result: \"{new_text}\".")


def decrypt(original_text, shift_amount):
    new_text = ""
    len_alphabet = len(alphabet)
    for char in original_text:
        if char in alphabet:
            char_index = alphabet.index(char)
            word_shift = alphabet[(char_index-shift_amount) % len_alphabet]
            new_text += word_shift
        else:
            new_text += char
    print(f"Here is the decoded result: \"{new_text}\".")


def caesar(pattern, original_text, shift_amount):
    if shift_amount < 0:
        shift_amount *= (-1)
    if pattern == "encode":
        encrypt(original_text, shift_amount)
    if pattern == "decode":
        decrypt(original_text, shift_amount)

while True:
