from datetime import datetime
from config import Config
from data_manager import DataManager
from flight_search import FlightSearch
from flight_data import FlightData
from notification_manager import NotificationManager

"""Main application class for finding flight deals."""

data_manager = DataManager()
flight_search = FlightSearch()
notification_manager = NotificationManager()
now_date = datetime.now().strftime("%Y-%m-%d")
departure_date = now_date


def update_city_codes():
    """Updates IATA codes for all cities in the database."""
    cities = data_manager.get_city_list()

    for city in cities:
        iata_code = flight_search.get_city_iata(city["city"])
        data_manager.update_city_iata(city["id"], iata_code)


def update_flight_prices(depart_date):
    """Updates flight prices for all destinations."""
    cities = data_manager.get_city_list(include_iata=True)

    for city in cities:
        flight_info = flight_search.search_flights(
            origin="TPE",
            destination=city["iataCode"],
            departure_date=depart_date
        )
        print(flight_info)

        data_manager.update_flight_data(
            city_id=city["id"],
            search_date=now_date,
            lastticketingdatetime=flight_info["lastTicketingDateTime"],
            departure_date=flight_info["departure_date"],
            departure_iata=flight_info["origin"],
            airline=flight_info["airline"],
            price=flight_info["price"],
            has_direct_flight=flight_info["hasDirectFlight"],
            flight_type=flight_info["flightType"],
            stopover_city=flight_info["stopoverCity"],
            stopover_count=flight_info["stopoverCount"]
        )


def check_price_alerts(recipient_email):
    """Checks for price alerts and sends notifications."""
    flight_data = data_manager.get_flight_data()

    for flight in flight_data:
        print(flight)
        lowest_price = float(flight.get('lowest_price', float('inf')))
        current_price = float(flight.get('flight_price', float('inf')))

        if current_price < lowest_price:
            flight_info = FlightData.from_dict(flight)
            message = (
                f"Low Price Alert!\n"
                f"Destination: {flight['city']}\n\n"
                f"{flight_info.format_message()}\n\n"
                f"Lower than expected price: {lowest_price:.2f} {Config.DEFAULT_CURRENCY}"
            )

            notification_manager.send_email(
                recipient_email=recipient_email,
                email_body=message
            )


def main():
    # departure_date = "2025-01-27"  # Could be parameterized
    now_date = datetime.now().strftime("%Y-%m-%d")
    departure_date = now_date

    # recipient_email = "123@gmail.com"  # Could be moved to config
    recipient_email_list = data_manager.get_receive_email_list()
    print(recipient_email_list)

    # Update city IATA codes
    update_city_codes()

    # Update flight prices
    update_flight_prices(departure_date)

    # Check for price alerts
    for recipient_email in recipient_email_list:
        check_price_alerts(recipient_email)


main()

import requests
from config import Config


class DataManager:
    """Manages interactions with the Google Sheet API."""

    def __init__(self):
        self.sheet_url = Config.SHEET_URL
        self.sheet_url_users = Config.SHEET_URL_USERS
        self.city_list = self.get_city_list()

    def get_receive_email_list(self):
        response = requests.get(url=self.sheet_url_users)
        response.raise_for_status()
        response_json = response.json()

        return [user["email"] for user in response_json["users"]]

    def get_city_list(self, include_iata=False):
        """
        Retrieves city list from the sheet.

        Args:
            include_iata: Whether to include IATA codes in the response

        Returns:
            List of city dictionaries
        """
        response = requests.get(url=self.sheet_url)
        response.raise_for_status()
        data = response.json()

        if include_iata:
            return [
                {
                    "id": item["id"],
                    "city": item["city"],
                    "iataCode": item["iataCode"]
                }
                for item in data["prices"]
            ]

        return [
            {
                "id": item["id"],
                "city": item["city"]
            }
            for item in data["prices"]
        ]

    def get_flight_data(self):
        """Retrieves complete flight data from the sheet."""
        response = requests.get(url=self.sheet_url)
        response.raise_for_status()

        return [
            {
                "id": item["id"],
                "lastTicketingDateTime": item["lastTicketingDateTime"],
                "departure_date": item["departDate"],
                "departure_iata": item["departIata"],
                "city": item["city"],
                "iata_code": item["iataCode"],
                "airline": item["airline"],
                "lowest_price": item["lowestPrice"],
                "flight_price": item["flightPrice"]
            }
            for item in response.json()["prices"]
        ]

    def update_city_iata(self, city_id, iata_code):
        """Updates IATA code for a specific city."""
        payload = {
            "price": {
                "iataCode": iata_code
            }
        }
        response = requests.put(
            url=f"{self.sheet_url}/{city_id}",
            json=payload
        )
        response.raise_for_status()

    def update_flight_data(self, city_id, search_date, lastticketingdatetime, departure_date,
                           departure_iata, airline, price, has_direct_flight=True,
                           flight_type="DIRECT", stopover_city=None, stopover_count=0):
        """Updates flight information for a specific city."""
        payload = {
            "price": {
                "searchDate": search_date,
                "lastTicketingDateTime": lastticketingdatetime,
                "departDate": departure_date,
                "departIata": departure_iata,
                "airline": airline,
                "flightPrice": str(price),
                "hasDirectFlight": has_direct_flight,
                "flightType": flight_type,
                "stopoverCity": stopover_city if stopover_city else "",
                "stopoverCount": stopover_count
            }
        }
        response = requests.put(
            url=f"{self.sheet_url}/{city_id}",
            json=payload
        )
        response.raise_for_status()

from config import Config


class FlightData:
    def __init__(self, price, origin, destination, last_ticket_date, departure_date,
                 airline=None, has_direct_flight=True, flight_type="DIRECT",
                 stopover_city=None, stopover_count=0):
        """Structures flight data in a consistent format."""
        self.price = price
        self.origin = origin
        self.destination = destination
        self.last_ticket_date = last_ticket_date
        self.departure_date = departure_date
        self.airline = airline
        self.has_direct_flight = has_direct_flight
        self.flight_type = flight_type
        self.stopover_city = stopover_city
        self.stopover_count = stopover_count

    def from_dict(data):
        """Creates FlightData instance from dictionary."""

        return FlightData(
            price=float(data.get("flight_price", 0)),
            origin=data.get("departure_iata", ""),
            destination=data.get("iata_code", ""),
            last_ticket_date=data.get("lastTicketingDateTime", ""),
            departure_date=data.get("departure_date", ""),
            airline=data.get("airline"),
            has_direct_flight=data.get("hasDirectFlight", True),
            flight_type=data.get("flightType", "DIRECT"),
            stopover_city=data.get("stopoverCity"),
            stopover_count=data.get("stopoverCount", 0)
        )

    def format_message(self):
        """Formats flight information as a readable message."""
        base_message = (
            f"Flight from {self.origin} to {self.destination}\n"
            f"Departure: {self.departure_date}\n"
            f"Price: {self.price:.2f} {Config.DEFAULT_CURRENCY}\n"
            f"Airline: {self.airline or 'Not specified'}\n"
            f"Last Ticketing Date: {self.last_ticket_date}\n"
            f"Flight Type: {self.flight_type}"
        )

        if not self.has_direct_flight and self.stopover_city:
            base_message += f"\nStopovers: {self.stopover_city} ({self.stopover_count} stops)"

        return base_message

import requests
from config import Config


class FlightSearch:
    """Handles interactions with the Amadeus Flight Search API."""

    def __init__(self):
        self.access_token = self._get_access_token()

    def _get_access_token(self) -> str:
        """Retrieves access token from Amadeus API."""
        headers = {
            "grant_type": "client_credentials",
            "client_id": Config.AMADEUS_KEY,
            "client_secret": Config.AMADEUS_SECRET
        }

        response = requests.post(
            url=f"{Config.AMADEUS_BASE_URL_V1}/{Config.AMADEUS_TOKEN_ENDPOINT}",
            data=headers
        )
        response.raise_for_status()
        token_data = response.json()

        return f"{token_data['token_type']} {token_data['access_token']}"

    def get_city_iata(self, city_name):
        """Retrieves IATA code for a given city name."""
        headers = {"Authorization": self.access_token}
        params = {"keyword": city_name}

        response = requests.get(
            url=f"{Config.AMADEUS_BASE_URL_V1}/{Config.AMADEUS_CITY_ENDPOINT}",
            headers=headers,
            params=params
        )
        response.raise_for_status()

        return response.json()['data'][0]['iataCode']

    def search_flights(self, origin, destination, departure_date,
                       adults=Config.DEFAULT_ADULTS,
                       currency=Config.DEFAULT_CURRENCY,
                       direct_only=True):
        """
        Searches for flights based on given parameters.

        Args:
            origin: Origin city IATA code
            destination: Destination city IATA code
            departure_date: Departure date in YYYY-MM-DD format
            adults: Number of adult passengers
            currency: Currency code for prices

        Returns:
            Dictionary containing flight information
        """
        headers = {"Authorization": self.access_token}

        params = {
            "originLocationCode": origin,
            "destinationLocationCode": destination,
            "departureDate": departure_date,
            "adults": adults,
            "nonStop": "true" if direct_only else "false",
            "currencyCode": currency,
            "max": Config.DEFAULT_MAX_RESULTS
        }

        response = requests.get(
            url=f"{Config.AMADEUS_BASE_URL_V2}/{Config.AMADEUS_FLIGHT_OFFERS_ENDPOINT}",
            headers=headers,
            params=params
        )
        response.raise_for_status()
        flights = response.json().get("data", [])

        if not flights:
            if direct_only:
                return self.search_flights(
                    origin=origin,
                    destination=destination,
                    departure_date=departure_date,
                    adults=adults,
                    currency=currency,
                    direct_only=False
                )
            return None

        flight_data = flights[0]
        segments = flight_data["itineraries"][0]["segments"]

        if len(segments) == 1:
            return {
                "price": flight_data["price"]["total"],
                "origin": segments[0]["departure"]["iataCode"],
                "destination": segments[0]["arrival"]["iataCode"],
                "lastTicketingDateTime": flight_data["lastTicketingDateTime"],
                "departure_date": segments[0]["departure"]["at"],
                "airline": flight_data["validatingAirlineCodes"][0],
                "hasDirectFlight": True,
                "flightType": "DIRECT",
                "stopoverCity": None,
                "stopoverCount": 0
            }

        stopovers = [segment["arrival"]["iataCode"] for segment in segments[:-1]]

        return {
            "price": flight_data["price"]["total"],
            "origin": segments[0]["departure"]["iataCode"],
            "destination": segments[-1]["arrival"]["iataCode"],
            "lastTicketingDateTime": flight_data["lastTicketingDateTime"],
            "departure_date": segments[0]["departure"]["at"],
            "airline": flight_data["validatingAirlineCodes"][0],
            "hasDirectFlight": False,
            "flightType": "INDIRECT",
            "stopoverCity": ",".join(stopovers),
            "stopoverCount": len(segments) - 1
        }

from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
import smtplib
from config import Config


class NotificationManager:
    """Handles email notifications for flight deals."""

    def __init__(self):
        self.sender_email = Config.MAIL_USERNAME
        self.sender_password = Config.MAIL_PASSWORD
        self.smtp_server = Config.SMTP_SERVER
        self.smtp_port = Config.SMTP_PORT

    def send_email(self, recipient_email, email_body, subject="Flight Deal Alert"):
        """
        Sends email notification with flight deal information.

        Args:
            recipient_email: Recipient's email address
            message: Email body content
            subject: Email subject line
        """
        msg = MIMEMultipart()
        msg['From'] = self.sender_email
        msg['To'] = recipient_email
        msg['Subject'] = subject
        msg.attach(MIMEText(email_body, 'plain', 'utf-8'))

        with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
            server.starttls()
            server.login(self.sender_email, self.sender_password)
            server.send_message(msg)
