import os
import pandas
import smtplib
import datetime as dt
import random

USERNAME = "aaabbbccc@gmail.com"
PASSWORD = "zzzxxxyyy"
WORKING_DICTORY = os.getcwd()
EMAIL_PATH = os.path.join(WORKING_DICTORY, "birthday-wisher-extrahard-start", "letter_templates")
BIRTHDAY_DATA_FILE = os.path.join(WORKING_DICTORY, "birthday-wisher-extrahard-start", "birthdays.csv")


def get_email_templates(email_path):
    return [file for file in os.listdir(email_path) if os.path.isfile(os.path.join(EMAIL_PATH, file))]


def get_random_template(email_path, email_templates_list):
    return os.path.join(email_path, random.choice(email_templates_list))


def personalize_email(mail_template, name):
    with open(mail_template, "r") as file:
        email_content = file.read()
    return email_content.replace("[NAME]", name)


def get_birthday_people(birthday_data_file):
    birthday_data = pandas.read_csv(birthday_data_file)
    now = dt.datetime.now()
    filtered_data = birthday_data[(birthday_data["month"] == now.month) & (birthday_data["day"] == now.day)]
    return filtered_data[["name", "email"]].to_dict(orient="records")


def send_email(recipient_email, email_body, username, password):
    with smtplib.SMTP("smtp.gmail.com") as connection:
        connection.starttls()
        connection.login(username, password=password)
        connection.sendmail(
            from_addr=username,
            to_addrs=recipient_email,
            msg=f"Subject: Happy Birthday!\n\n{email_body}"
        )


def process_birthday_email(birthday_people, template_list, username, password):
    mail_template = get_random_template(EMAIL_PATH, template_list)
    content = personalize_email(mail_template, birthday_people['name'])
    send_email(birthday_people['email'], content, username, password)


# <!---------------- main process ---------------------->


all_email_template_list = get_email_templates(EMAIL_PATH)
birthday_list = get_birthday_people(BIRTHDAY_DATA_FILE)

for people in birthday_list:
    process_birthday_email(people, all_email_template_list, USERNAME, PASSWORD)
