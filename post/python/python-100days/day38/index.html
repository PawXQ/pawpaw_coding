import requests
from datetime import datetime

# NUTRITIONIX CONFIGURATION
NUTRITIONIX_API_URL = "https://trackapi.nutritionix.com"
NUTRITIONIX_API_ENDPOINT = "/v2/natural/exercise"
NUTRITIONIX_API_ID = "123456"
NUTRITIONIX_API_KEY = "123456"

# USER INFO
GENDER = "male"
WEIGHT_KG = 100
HEIGHT_CM = 200
AGE = 30

# SHEET_API
SHEET_API_URL = "https://api.sheety.co/4f859e749ece55464716d3cf442c5497/myWorkouts/workouts"
TOKEN = "123456"


def get_exercise_data(exercise_input):
    headers = {
        "x-app-id": NUTRITIONIX_API_ID,
        "x-app-key": NUTRITIONIX_API_KEY,
    }

    parameters = {
        "query": exercise_input,
        "gender": GENDER,
        "weight_kg": WEIGHT_KG,
        "height_cm": HEIGHT_CM,
        "age": AGE
    }

    response = requests.post(
        url=f"{NUTRITIONIX_API_URL}{NUTRITIONIX_API_ENDPOINT}",
        headers=headers,
        json=parameters
    )
    response.raise_for_status()
    return response.json()


def prepare_workout_data(exercise_data):
    now = datetime.now()

    return {
        "workout": {
            "date": now.strftime("%d/%m/%Y"),
            "time": now.strftime("%H:%M:%S"),
            "exercise": exercise_data["exercises"][0]["name"],
            "duration": exercise_data["exercises"][0]["duration_min"],
            "calories": exercise_data["exercises"][0]["nf_calories"]
        }
    }


def save_to_sheet(workout_data):
    headers = {
        "Authorization": f"Bearer {TOKEN}"
    }
    response = requests.post(url=SHEET_API_URL, json=workout_data, headers=headers)
    return response.text


# Get exercise input from user
exercise_input = input("Tell me which exercises you did: ")

# Process exercise data
exercise_data = get_exercise_data(exercise_input)

# Prepare workout data
workout_data = prepare_workout_data(exercise_data)

# Save to sheet and print response
print(save_to_sheet(workout_data))
