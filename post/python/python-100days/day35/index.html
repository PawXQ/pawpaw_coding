import requests
from twilio.rest import Client

FROM_NUMBER = os.environ.get("FROM_NUMBER")
TO_NUMBER = os.environ.get("TO_NUMBER")


API_URL = "https://api.openweathermap.org/data/2.5/forecast"


API_KEY = os.environ.get("OWM_API_KEY")
LAT = os.environ.get("LAT")
LON = os.environ.get("LON")


ACCOUNT_SID = os.environ.get("TWILIO_ACCOUNT_SID")
AUTH_TOKEN = os.environ.get("TWILIO_AUTH_TOKEN")
client = Client(ACCOUNT_SID, AUTH_TOKEN)


PARAMETER = {
    "lat": LAT,
    "lon": LON,
    "appid": API_KEY,
    "cnt": "8",
}


def send_sms(body, from_number, to_number, account_sid, auth_token):
    client = Client(account_sid, auth_token)
    message = client.messages.create(
        body=body,
        from_=from_number,
        to=to_number
    )
    print(message.sid)
    print(message.status)


response = requests.get(url=API_URL, params=PARAMETER)
response.raise_for_status()
weather_data = response.json()


weather_data_id_list = [weather_data_list["weather"][0]["id"] for weather_data_list in weather_data["list"]]
if any(i < 800 for i in weather_data_id_list):
    notification_body = "It's going to rain today. Remember to bring an ☂️."
    send_sms(
        body=notification_body,
        account_sid=ACCOUNT_SID,
        auth_token=AUTH_TOKEN,
        from_number=FROM_NUMBER,
        to_number=TO_NUMBER
    )
